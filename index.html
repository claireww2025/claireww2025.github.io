            function getGaokaoStatus(uni) {

                if (!uni.undergraduate || !uni.undergraduate.gaokao) return { text: 'N/A', color: 'gray' };

                switch (uni.undergraduate.gaokao) {

                    case 'yes': return { text: '接受高考', color: 'green' };

                    case 'no': return { text: '不接受高考', color: 'red' };

                    case 'no_foundation': return { text: '需读预科', color: 'yellow' };

                    case 'no_transfer': return { text: '需转学', color: 'blue' };

                    default: return { text: 'N/A', color: 'gray' };

                }

            }

            function applyFilters() {

                const degreeLevel = degreeLevelFilter.value;

                const undergradPath = undergradPathFilter.value;

                const qsRank = qsRankFilter.value;

                const searchTerm = searchFilter.value.toLowerCase();

                const hasScholarship = scholarshipFilter.checked;


                undergradPathFilterContainer.classList.toggle('hidden', degreeLevel !== 'undergraduate');


                let filteredData = universityData;


                if (hasScholarship) {

                    filteredData = filteredData.filter(uni => uni.scholarship);

                }


                if (qsRank !== 'all') {

                    const [min, max] = qsRank.split('-').map(Number);

                    filteredData = filteredData.filter(uni => uni.qsRank >= min && uni.qsRank <= max);

                }


                if (searchTerm) {

                    filteredData = filteredData.filter(uni => uni.name.toLowerCase().includes(searchTerm));

                }


                if (degreeLevel === 'undergraduate' && undergradPath !== 'all') {

                    filteredData = filteredData.filter(uni => {

                        if (!uni.undergraduate || !uni.undergraduate.gaokao) return false;

                        const gaokaoType = uni.undergraduate.gaokao;

                        if (undergradPath === 'gaokao') return gaokaoType === 'yes';

                        if (undergradPath === 'foundation') return gaokaoType.includes('foundation') || gaokaoType.includes('transfer');

                        if (undergradPath === 'international') return gaokaoType === 'no' || gaokaoType.includes('foundation') || gaokaoType.includes('transfer');

                        return true;

                    });

                }

                

                renderCards(filteredData);

            }
